sudo apt update
sudo apt install docker.io -y

docker volume create mysql-data

docker run -d \
--name mysql-db \
--restart unless-stopped \
-v mysql-data:/var/lib/mysql \
-e MYSQL_ROOT_PASSWORD=Root@123 \
-e MYSQL_DATABASE=projectdb \
-p 3306:3306 \
mysql:8
